struct mi_visitor {
  void operator()(const int & i) { std::cout << "int: " << i << "\n"; }
  void operator()(const double & d) { std::cout << "double: " << d << "\n"; }
};

void f() {
  std::variant<int,float> v;

  v = 2.0;
  std::visit(mi_visitor{}, v);

  v = 5;
  std::visit(mi_visitor{}, v);
}
\end{lstlisting}
