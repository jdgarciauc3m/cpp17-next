\begin{frame}[t,fragile]{\textbf{string\_view}}
\begin{itemize}
  \item Define un objeto que se refiere a una secuencia de caracteres
        constante.
    \begin{itemize}
      \item Tiene una interfaz muy similar a \cppid{std::string}.
    \end{itemize}
\end{itemize}

std::string_view sv1 = "hola";

\end{frame}
